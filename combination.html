var count = 0

func comb(n, k int, arr []int) {
    count++
    fmt.Println("--", count, "--")
    for i := n; i >= k; i-- {
        arr[k-1] = i
        if k > 1 {
            comb(n-1, k-1, arr)
        } else {
            fmt.Println(arr)
        }
    }
    count--
}
